214 // Functions that fills in the SIDIS hard cross sections on two
215 // different grids.
216 SidisObjects InitializeSIDISpol(Grid const& gx, Grid const& gz, std::vector<double> const& Thresholds, std::vector<int> exclude = {})
217 {
218 report("Initializing SIDIS longitudinally polarised hard cross sections... ");
219 Timer t;
220
221 // Compute initial and final number of active flavours according
222 // to the vector of thresholds (it assumes that the thresholds
223 // vector entries are ordered).
224 int nfi = 0;
225 int nff = Thresholds.size();
226 for (auto const& v : Thresholds)
227 if (v <= 0)
228 nfi++;
229
230 // Define object of the structure containing the DglapObjects.
231 SidisPolObjects SidisObj;
232
233 // ====================================================
234 // Compute SIDIS partonic cross sections.
235 // Expressions taken from Appendix C of hep-ph/9711387.
236 // ====================================================
237 // LO contribution.
238 const Operator odeltax{gx, delta{}};
239 const Operator odeltaz{gz, delta{}};
240
241 if (std::find(exclude.begin(), exclude.end(), 0) == exclude.end()) SidisObj.G10qq.AddTerm({1, odeltax, odeltaz});
242
243 // NLO contributions
244 // G1
245 const Operator oD0x{gx, Dn{0}};
246 const Operator oD1x{gx, Dn{1}};
247 const Operator oD0z{gz, Dn{0}};
248 const Operator oD1z{gz, Dn{1}};
249
250 const double LLqq = - 16 * CF;
251 const double LSqq = 4 * CF;
252 const double SLqq = 4 * CF;
253 const double SSqq = 4 * CF;
254 const double K1qq = 4 * CF;
255 const double K2qq = 4 * CF;
256
257 const Operator orlqqx{gx, rlqq{}};
258 const Operator orsqqx{gx, rsqq{}};
259 const Operator or11qqx{gx, r11qq{}};
260 const Operator or21qqx{gx, r21qq{}};
261 const Operator olrqqz{gz, lrqq{}};
262 const Operator osrqqz{gz, srqq{}};
263 const Operator or12qqz{gz, r12qq{}};
264 const Operator or22qqz{gz, r22qq{}};
265
266 if (std::find(exclude.begin(), exclude.end(), 1) == exclude.end()) SidisObj.G11qq.AddTerm({LLqq, odeltax, odeltaz});
267 if (std::find(exclude.begin(), exclude.end(), 2) == exclude.end()) SidisObj.G11qq.AddTerm({LSqq, odeltax, oD1z });
268 if (std::find(exclude.begin(), exclude.end(), 3) == exclude.end()) SidisObj.G11qq.AddTerm({1, odeltax, olrqqz });
269 if (std::find(exclude.begin(), exclude.end(), 4) == exclude.end()) SidisObj.G11qq.AddTerm({SLqq, oD1x, odeltaz});
270 if (std::find(exclude.begin(), exclude.end(), 5) == exclude.end()) SidisObj.G11qq.AddTerm({SSqq, oD0x, oD0z });
271 if (std::find(exclude.begin(), exclude.end(), 6) == exclude.end()) SidisObj.G11qq.AddTerm({1, oD0x, osrqqz });
272 if (std::find(exclude.begin(), exclude.end(), 7) == exclude.end()) SidisObj.G11qq.AddTerm({1, orlqqx, odeltaz});
273 if (std::find(exclude.begin(), exclude.end(), 8) == exclude.end()) SidisObj.G11qq.AddTerm({1, orsqqx, oD0z });
274 if (std::find(exclude.begin(), exclude.end(), 9) == exclude.end()) SidisObj.G11qq.AddTerm({K1qq, or11qqx, or12qqz});
275 if (std::find(exclude.begin(), exclude.end(), 10) == exclude.end()) SidisObj.G11qq.AddTerm({K2qq, or21qqx, or22qqz});
276
277 const double K1gq = 2 * CF;
278 const double K2gq = - 4 * CF;
279
280 const Operator olrgqz{gz, lrgq{}};
281 const Operator osrgqz{gz, {}};
282 const Operator or11gqx{gx, r11gq{}};
283 const Operator or12gqz{gz, r12gq{}};
284 const Operator or21gqx{gx, r21gq{}};
285 const Operator or22gqz{gz, r22gq{}};
286
287 if (std::find(exclude.begin(), exclude.end(), 11) == exclude.end()) SidisObj.G11gq.AddTerm({1, odeltax, olrgqz });
288 if (std::find(exclude.begin(), exclude.end(), 12) == exclude.end()) SidisObj.G11gq.AddTerm({1, oD0x, osrgqz });
289 if (std::find(exclude.begin(), exclude.end(), 13) == exclude.end()) SidisObj.G11gq.AddTerm({K1gq, or11gqx, or12gqz});
290 if (std::find(exclude.begin(), exclude.end(), 14) == exclude.end()) SidisObj.G11gq.AddTerm({K2gq, or21gqx, or22gqz});
291
292 const double K1qg = 1;
293
294 const Operator orlqgx{gx, rlqg{}};
295 const Operator orsqgx{gx, rsqg{}};
296 const Operator or11qgx{gx, r11qg{}};
297 const Operator or12qgz{gz, r12qg{}};
298
299 if (std::find(exclude.begin(), exclude.end(), 15) == exclude.end()) SidisObj.G11qg.AddTerm({1, orlqgx, odeltaz});
300 if (std::find(exclude.begin(), exclude.end(), 16) == exclude.end()) SidisObj.G11qg.AddTerm({1, orsqgx, oD0z });
301 if (std::find(exclude.begin(), exclude.end(), 17) == exclude.end()) SidisObj.G11qg.AddTerm({K1qg, or11qgx, or12qgz});
302
303 // ====================================================
304 // Approximated NNLO corrections derived from threshold
305 // resummation. They only contribute to the qq channel of F2.
306 // Expressions taken from Appendix B of arXiv:2109.00847.
307 // ====================================================
308 // Additional singular terms
309 const Operator oD2x{gx, Dn{2}};
310 const Operator oD3x{gx, Dn{3}};
311 const Operator oD2z{gz, Dn{2}};
312 const Operator oD3z{gz, Dn{3}};
313
314 // Non-singular (next-to-leading power) terms
315 const Operator ol1x{gx, ln{1}};
316 const Operator ol2x{gx, ln{2}};
317 const Operator ol3x{gx, ln{3}};
318 const Operator ol1z{gz, ln{1}};
319 const Operator ol2z{gz, ln{2}};
320 const Operator ol3z{gz, ln{3}};
321
322 // Constant function
323 const Operator oonex{gx, one{}};
324 const Operator oonez{gz, one{}};
325
326 // Overall constant (16 = 4^2 due to the different normalisation
327 // of the expansion parameter)
328 const double ovc = 16. * CF;
329
330 // Loop over nf
331 for (int nf = nfi; nf <= nff; nf++)
332 {
333 DoubleObject<Operator> cnf{};
334
335 // CF and leading-power terms (Eq. 62)
336 const double c1 = ovc * CF / 2.;
337 if (std::find(exclude.begin(), exclude.end(), 18) == exclude.end()) cnf.AddTerm({c1, odeltax, oD3z});
338 if (std::find(exclude.begin(), exclude.end(), 19) == exclude.end()) cnf.AddTerm({c1, oD3x, odeltaz});
339
340 const double c2 = ovc * CF * 3. / 2.;
341 if (std::find(exclude.begin(), exclude.end(), 20) == exclude.end()) cnf.AddTerm({c2, oD0x, oD2z});
342 if (std::find(exclude.begin(), exclude.end(), 21) == exclude.end()) cnf.AddTerm({c2, oD2x, oD0z});
343 if (std::find(exclude.begin(), exclude.end(), 22) == exclude.end()) cnf.AddTerm({2 * c2, oD1x, oD1z});
344
345 const double c3 = - ovc * CF * ( 4. + Pi2 / 3. );
346 if (std::find(exclude.begin(), exclude.end(), 23) == exclude.end()) cnf.AddTerm({c3, oD0x, oD0z});
347 if (std::find(exclude.begin(), exclude.end(), 24) == exclude.end()) cnf.AddTerm({c3, odeltax, oD1z});
348 if (std::find(exclude.begin(), exclude.end(), 25) == exclude.end()) cnf.AddTerm({c3, oD1x, odeltaz});
349
350 const double c4 = ovc * CF * 2. * zeta3;
351 if (std::find(exclude.begin(), exclude.end(), 26) == exclude.end()) cnf.AddTerm({c4, odeltax, oD0z});
352 if (std::find(exclude.begin(), exclude.end(), 27) == exclude.end()) cnf.AddTerm({c4, oD0x, odeltaz});
353
354 const double c5 = ovc * CF * ( 511. / 64. - 15. * zeta3 / 4. + 29. * Pi2 / 48. - 7. * Pi2 * Pi2 / 360. );
355 if (std::find(exclude.begin(), exclude.end(), 28) == exclude.end()) cnf.AddTerm({c5, odeltax, odeltaz});
356
357 // CF and next-to-leading-power terms (Eq. 63)
358 const double c6 = - ovc * CF * 3. / 2.;
359 if (std::find(exclude.begin(), exclude.end(), 29) == exclude.end()) cnf.AddTerm({c6, oD2x, oonez});
360 if (std::find(exclude.begin(), exclude.end(), 30) == exclude.end()) cnf.AddTerm({c6, oonex, oD2z});
361 if (std::find(exclude.begin(), exclude.end(), 31) == exclude.end()) cnf.AddTerm({2 * c6, oD1x, ol1z});
362 if (std::find(exclude.begin(), exclude.end(), 32) == exclude.end()) cnf.AddTerm({2 * c6, ol1x, oD1z});
363 if (std::find(exclude.begin(), exclude.end(), 33) == exclude.end()) cnf.AddTerm({c6, oD0x, ol2z});
364 if (std::find(exclude.begin(), exclude.end(), 34) == exclude.end()) cnf.AddTerm({c6, ol2x, oD0z});
365
366 const double c7 = - ovc * CF / 2.;
367 if (std::find(exclude.begin(), exclude.end(), 35) == exclude.end()) cnf.AddTerm({c7, odeltax, ol3z});
368 if (std::find(exclude.begin(), exclude.end(), 36) == exclude.end()) cnf.AddTerm({c7, ol3x, odeltaz});
369
370 // CA terms (Eq. 64)
371 const double c8 = - ovc * CA * 11. / 24.;
372 if (std::find(exclude.begin(), exclude.end(), 37) == exclude.end()) cnf.AddTerm({c8, odeltax, oD2z});
373 if (std::find(exclude.begin(), exclude.end(), 38) == exclude.end()) cnf.AddTerm({c8, oD2x, odeltaz});
374 if (std::find(exclude.begin(), exclude.end(), 39) == exclude.end()) cnf.AddTerm({2 * c8, oD0x, oD1z});
375 if (std::find(exclude.begin(), exclude.end(), 40) == exclude.end()) cnf.AddTerm({2 * c8, oD1x, oD0z});
376
377 const double c9 = ovc * CA * ( 67. / 36. - Pi2 / 12. );
378 if (std::find(exclude.begin(), exclude.end(), 41) == exclude.end()) cnf.AddTerm({c9, oD0x, oD0z});
379 if (std::find(exclude.begin(), exclude.end(), 42) == exclude.end()) cnf.AddTerm({c9, odeltax, oD1z});
380 if (std::find(exclude.begin(), exclude.end(), 43) == exclude.end()) cnf.AddTerm({c9, oD1x, odeltaz});
381
382 const double c10 = ovc * CA * ( 7. * zeta3 / 4. + 11. * Pi2 / 72. - 101. / 54. );
383 if (std::find(exclude.begin(), exclude.end(), 44) == exclude.end()) cnf.AddTerm({c10, odeltax, oD0z});
384 if (std::find(exclude.begin(), exclude.end(), 45) == exclude.end()) cnf.AddTerm({c10, oD0x, odeltaz});
385
386 const double c11 = ovc * CA * ( 43. * zeta3 / 12. + 17. * Pi2 * Pi2 / 720. - 1535. / 192. - 269. * Pi2 / 432. );
387 if (std::find(exclude.begin(), exclude.end(), 46) == exclude.end()) cnf.AddTerm({c11, odeltax, odeltaz});
388
389 // nf terms (Eq. 65)
390 const double c12 = ovc * nf / 12.;
391 if (std::find(exclude.begin(), exclude.end(), 47) == exclude.end()) cnf.AddTerm({c12, odeltax, oD2z});
392 if (std::find(exclude.begin(), exclude.end(), 48) == exclude.end()) cnf.AddTerm({c12, oD2z, odeltaz});
393 if (std::find(exclude.begin(), exclude.end(), 49) == exclude.end()) cnf.AddTerm({2 * c12, oD0x, oD1z});
394 if (std::find(exclude.begin(), exclude.end(), 50) == exclude.end()) cnf.AddTerm({2 * c12, oD1x, oD0z});
395
396 const double c13 = - ovc * nf * 5. / 18.;
397 if (std::find(exclude.begin(), exclude.end(), 51) == exclude.end()) cnf.AddTerm({c13, oD0x, oD0z});
398 if (std::find(exclude.begin(), exclude.end(), 52) == exclude.end()) cnf.AddTerm({c13, odeltax, oD1z});
399 if (std::find(exclude.begin(), exclude.end(), 53) == exclude.end()) cnf.AddTerm({c13, oD1x, odeltaz});
400
401 const double c14 = ovc * nf * ( 7. / 27. - Pi2 / 36. );
402 if (std::find(exclude.begin(), exclude.end(), 54) == exclude.end()) cnf.AddTerm({c14, odeltax, oD0z});
403 if (std::find(exclude.begin(), exclude.end(), 55) == exclude.end()) cnf.AddTerm({c14, oD0x, odeltaz});
404
405 const double c15 = ovc * nf * ( zeta3 / 6. + 19. * Pi2 / 216. + 127. / 96. );
406 if (std::find(exclude.begin(), exclude.end(), 56) == exclude.end()) cnf.AddTerm({c15, odeltax, odeltaz});
407
408 SidisObj.G12qq.insert({nf, cnf});
409 }
410 t.stop();
411
412 return SidisPolObj;
413 }
